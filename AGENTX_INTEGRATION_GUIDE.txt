import { useEffect } from 'react';

export default function DashboardLayout({ children }: { children: React.ReactNode }) {
  useEffect(() => {
    // Initialize AgentX chatbot - Dashboard only
    const initAgentX = () => {
      // Create chat bubble container
      const chatBubbleRoot = document.getElementById('chatBubbleRoot');
      if (!chatBubbleRoot) {
        const root = document.body.appendChild(document.createElement('div'));
        root.setAttribute('id', 'chatBubbleRoot');
      }

      // Set AgentX credentials
      window.agx = '698a2150adfb07e4322f8b14DB3zn76SnuxyDvRzIQIdUA==|SXumI/A4zPRadJpcBuLNPrPQ9nR9nHNmlJp5z+xR3Ts=';

      // Load AgentX script
      const script = document.createElement('script');
      script.src = 'https://storage.googleapis.com/agentx-cdn-01/agentx-chat.js';
      script.defer = true;
      document.body.appendChild(script);
    };

    initAgentX();

    // Cleanup: Remove AgentX when leaving dashboard
    return () => {
      const chatBubbleRoot = document.getElementById('chatBubbleRoot');
      if (chatBubbleRoot) {
        chatBubbleRoot.remove();
      }
      // Remove AgentX script
      const scripts = document.querySelectorAll('script[src="https://storage.googleapis.com/agentx-cdn-01/agentx-chat.js"]');
      scripts.forEach(script => script.remove());
    };
  }, []);

  return (
    <div className="flex h-screen bg-background">
      {/* Your existing layout code */}
      {children}
    </div>
  );
}

import { useEffect } from 'react';

function AgentXLoader() {
  useEffect(() => {
    const currentPath = window.location.pathname;
    const isDashboard = currentPath.includes('/dashboard');

    if (isDashboard) {
      // Create chat bubble root
      let chatBubbleRoot = document.getElementById('chatBubbleRoot');
      if (!chatBubbleRoot) {
        chatBubbleRoot = document.createElement('div');
        chatBubbleRoot.setAttribute('id', 'chatBubbleRoot');
        document.body.appendChild(chatBubbleRoot);
      }

      // Initialize AgentX
      window.agx = '698a2150adfb07e4322f8b14DB3zn76SnuxyDvRzIQIdUA==|SXumI/A4zPRadJpcBuLNPrPQ9nR9nHNmlJp5z+xR3Ts=';

      // Load script if not already loaded
      if (!document.querySelector('script[src="https://storage.googleapis.com/agentx-cdn-01/agentx-chat.js"]')) {
        const script = document.createElement('script');
        script.src = 'https://storage.googleapis.com/agentx-cdn-01/agentx-chat.js';
        script.defer = true;
        document.body.appendChild(script);
      }
    }
  }, []);

  return null;
}

TROUBLESHOOTING
===============

Issue: Chat bubble appears on all pages
Solution: 
- Use Method 1 (DashboardLayout) for dashboard-only display
- Ensure the cleanup function is running correctly
- Clear browser cache (Ctrl+Shift+Del)

Issue: Chat bubble not appearing on dashboard
Solution:
- Check if AgentX script loaded (Console → Network tab)
- Verify credentials are correct
- Check browser console for errors (F12)
- Make sure JavaScript is enabled

Issue: Chat bubble appears multiple times
Solution:
- Check cleanup function is removing duplicate instances
- Clear localStorage and cache
- Restart development server

Issue: Chat bubble blocks content
Solution:
- Add CSS to main.css to reposition:
  #chatBubbleRoot {
    z-index: 999;
    position: fixed;
    bottom: 20px;
    right: 20px;
  }
